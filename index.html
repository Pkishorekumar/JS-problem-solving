1:-Find the Smaller Angle

     var Minimal_Angle = (h, m) => 
{
  var a=(360/12)*(h+(m/60));
  var b=(360/60)*m;
  
  var ang=Math.abs(a-b);
  return Math.min(ang,360-ang);
};

3:-Perfect Number Check.

      var Perfect_Check = (N) => 
{
  var a=1;
  for( var i=0;i< N;i++){
    if(N%i===0){
      a += i;
    }
  }
   if(a===N){
     return "YES";
   }
     else{
       return "NO"
     }
};

4:-Reverse a Number.

       var Reverse_Number = (N) => 
{
  var a=0;
  while(N>0){
    a=a*10+(N%10);
    N=Math.floor(N/10);
  }
  return a;
};
